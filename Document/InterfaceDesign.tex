\section{Interface design}
In this section all the internal and external interfaces are listed and described. A Figure of all components and interfaces can be seen in Figure \ref{fig:ComponentsAndInterfaces}

\subsection{Power interfaces}

\subsubsection{P\_1.1}
\begin{enumerate}[label = \bfseries ID-1.\arabic*]
\item The interface delivers power from the F-16 power supply to the SPS\_PCU. \label{ID-11}
\item The voltage is 115 VAC 400 Hz. \label{ID-12}
\item The SPS\_PCU will consume up to 250W.\label{ID-13}
\end{enumerate}

\subsubsection{P\_2.X}
\begin{enumerate}[label = \bfseries ID-2.\arabic*]
\item All the interfaces named P\_2.1-2.4 are power interfaces from the F-16 power supply to each DSS. \label{ID-21}
\item The voltage is 115 VAC 400 Hz. \label{ID-22}
\item Each DSS will consume up to 126W.\label{ID-23}
\end{enumerate}

\subsubsection{P\_3.X}
\begin{enumerate}[label = \bfseries ID-3.\arabic*]
\item All the interfaces named P\_3.1-3.4 are power interfaces from the SPS\_PCU to each DSS. \label{ID-31}
\item The voltage is 28V DC. \label{ID-32}
\item Each DSS consumes 3W.\label{ID-33}
\end{enumerate}

\subsubsection{P\_4.1}
\begin{enumerate}[label = \bfseries ID-4.\arabic*]
\item The interface delivers power from the SPS\_PCU to the MWS. \label{ID-41}
\item The voltage is 28V DC. \label{ID-42}
\item The MWS will consume up to 85W.\label{ID-43}
\end{enumerate}

\subsubsection{P\_5.1}
\begin{enumerate}[label = \bfseries ID-5.\arabic*]
\item The interface delivers power from the F-16 PCU to the SPS\_CU. \label{ID-51}
\item The voltage is 28V DC. \label{ID-52}
\item The SPS\_CU consumes up to 10W.\label{ID-53}
\end{enumerate}

\subsection{Communication and control interfaces}

\subsubsection{CC\_1.X}
\begin{enumerate}[label = \bfseries ID-6.\arabic*]
\item The interfaces CC\_1.1-1.4 are used for control of each DSS by the MWS. \label{ID-61}
\end{enumerate}

\subsubsection{CC\_2.X}
\begin{enumerate}[label = \bfseries ID-7.\arabic*]
\item The interfaces CC\_2.1-2.8 are used for igniting payloads by the DSS'.\label{ID-71}
\end{enumerate}

\subsubsection{CC\_3.1}
\begin{enumerate}[label = \bfseries ID-8.\arabic*]
\item The interface CC\_3.1 are used to turn on/off the power to the pod from the cockpit unit.\label{ID-81}
\end{enumerate}


\subsection{Multi connect bus interfaces}

\subsubsection{MCB\_1.1}
\begin{enumerate}[label = \bfseries ID-9.\arabic*]
\setcounter{enumi}{0}

\item The interface is of type real-time data transfer and will have a maximal time of providing measured data in under 5 ms. (SR-19)\label{ID-91}

\item The interface sets 3 different modes up to be chosen; automatic, semi-automatic and manual. (SR-1,2,3)\label{ID-92}

\item The interface provides two angles for each threat detected -- the first is 180 degree vertical, the second is 360 degree horizontal. (SR-6, SR-7 (SR-9?))\label{ID-93}

\item The interface provides a list of patterns the flares can be dispatched in.\label{ID-94}

\item The interface provides a launch message for manual mode and semi-automatic mode to launch flares in the selected pattern.\label{ID-95}
\end{enumerate}
